\hlcom{## @knitr example_IMU}

\hlcom{# Load IMU data}
\hlkwd{data}\hlstd{(imu6,} \hlkwc{package} \hlstd{=} \hlstr{"imudata"}\hlstd{)}

\hlcom{# Trim data}
\hlstd{yt} \hlkwb{=} \hlstd{imu6[,}\hlnum{1}\hlstd{]} \hlopt{-} \hlkwd{mean}\hlstd{(imu6[,}\hlnum{1}\hlstd{])}
\hlstd{xt} \hlkwb{=} \hlkwd{rep}\hlstd{(}\hlnum{NA}\hlstd{,} \hlkwd{floor}\hlstd{(}\hlkwd{length}\hlstd{(yt)}\hlopt{/}\hlnum{100}\hlstd{))}
\hlstd{counter} \hlkwb{=} \hlnum{0}
\hlkwa{for} \hlstd{(i} \hlkwa{in} \hlnum{1}\hlopt{:}\hlkwd{length}\hlstd{(yt))\{}
  \hlkwa{if} \hlstd{(i} \hlopt{%%} \hlnum{100} \hlopt{==} \hlnum{0}\hlstd{)\{}
    \hlstd{counter} \hlkwb{=} \hlstd{counter} \hlopt{+} \hlnum{1}
    \hlstd{inter_max} \hlkwb{=} \hlkwd{max}\hlstd{(yt[((i}\hlopt{-}\hlnum{100}\hlstd{)}\hlopt{+}\hlnum{1}\hlstd{)}\hlopt{:}\hlstd{i])}
    \hlstd{inter_min} \hlkwb{=} \hlkwd{min}\hlstd{(yt[((i}\hlopt{-}\hlnum{100}\hlstd{)}\hlopt{+}\hlnum{1}\hlstd{)}\hlopt{:}\hlstd{i])}

    \hlkwa{if} \hlstd{(inter_max} \hlopt{>} \hlkwd{abs}\hlstd{(inter_min))\{}
      \hlstd{xt[counter]} \hlkwb{=} \hlstd{inter_max}
    \hlstd{\}}\hlkwa{else}\hlstd{\{}
      \hlstd{xt[counter]} \hlkwb{=} \hlstd{inter_min}
    \hlstd{\}}
  \hlstd{\}}
\hlstd{\}}
\hlstd{xt} \hlkwb{=} \hlstd{xt} \hlopt{+} \hlkwd{mean}\hlstd{(imu6[,}\hlnum{1}\hlstd{])}
